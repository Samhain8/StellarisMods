# If the ambassador is actually a spymaster, this will substitute the title in a subtle way
defined_text = {
	name = GetAmbassadorOrNot
	text = {
		trigger = {
			NOT = {
				has_envoy_task = {
					task = spy_network
				}
			}
		}
		localization_key = STRING_ambassador
	}
	text = {
		trigger = {
			has_envoy_task = {
				task = spy_network
			}
		}
		localization_key = STRING_fake_ambassador
	}
}
